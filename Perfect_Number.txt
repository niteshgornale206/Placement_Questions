import java.util.Scanner;

public class Perfect {

	public static void main(String[] args) {
		Scanner sc= new Scanner(System.in);
		System.out.print("enter the number to check it is a perfect number: ");
		int n=sc.nextInt();
		int sum=0;
		for(int i=1;i<=n/2;i++) {
			if (n%i==0) {
				sum+=i;
				
			}
		}
		if(sum==n) System.out.println("The given number is a perfect number");
		else System.out.println("Not a perfect number");

	}

}
