import java.util.Arrays;

public class Shift {
	static void swap(int i,int j,int[] arr) {
		int temp=arr[i];
		arr[i]=arr[j];
		arr[j]=temp;
	}

	public static void main(String[] args) {
		
		int arr[]= {10,21,23,10,44,10,10,23,10,365,8955,10,256,10};
		int i=0;
		int j=arr.length-1;
		while(i<j)
		{
			if(arr[i]==10)
			{
				swap(i,j,arr);
				j--;
				
			}
			else
			{
				i++;
			}
		}

		System.out.println(Arrays.toString(arr));
	}
	
}
